-- ========================================
-- Название задачи: Компании, летавшие на Boeing
-- Описание:     Вывести названия компаний, которые
--               совершали перелеты на самолетах Boeing
-- Автор:        Е.Заржицкая
-- Дата:         2026-07-19
-- ========================================

SELECT DISTINCT Company.name
FROM Company
INNER JOIN Trip ON Trip.company = Company.id
WHERE Trip.plane = 'Boeing';
